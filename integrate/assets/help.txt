USAGE

    integ master [-n NUMBER] [-s VALUE] [-p PORT] [-q]
    integ slave [-p PORT] [-m HOST] [-q]
    integ -h

OPTIONS

    -n|--number NUMBER
    The number of expected slaves.
    default: 1

    -s|--step VALUE
    The step size for trapeze calculus.
    default: 0.0001

    -p|--port PORT
    The port which master listens to.
    default: 8989

    -m|--master HOST
    The master's host.
    default: localhost

    -q|--quiet
    Do not show output. Only errors and
    timing are showed.

    -h|--help
    Show this message.
